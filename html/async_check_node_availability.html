  async check_node_availability ( page, node_identity ){

    //to ensure availability nodes before processing  
    echo( `check if node ${node_identity} are available` );    

    var el_img_title = false;
    var n_eit = 1;
                    
    do{
      el_img_title = await page.waitForSelector( node_identity, { timeout:20000 } )
        .then( 
          ( resolved )=>{

            echo( `element #img_title found in ${n_eit} attempts` );

            return resolved;

          }
        )
        .catch( 
          async ( err )=>{

            if( err ){

              return false;

            }

          } 
        );
        await page.waitForTimeout( 1000 );
        n_eit++;
    }while( !el_img_title );
      
    return el_img_title;
      
  }
