  async writing_post( page, content ){
    
    // write posting
    
    echo( "writing post" );
    
    echo( 'waiting for selector iframe' );
    
    await page.waitForSelector( 'iframe' );
    
    echo( "get element iframe" );
    
    const elementHandle = await page.$( 'iframe' );
    
    echo( "get iframe content" );
    
    const framec = await elementHandle.contentFrame();
    
    echo( "pass variable content into tinymce editor box" );
    
    var eval_content = await framec.evaluate( ( content ) => {
    
      document.getElementById( "tinymce" ).innerHTML = "";
      
      document.getElementById( "tinymce" ).innerHTML = content;
      
      return true;
      
    }, content );
    
    return eval_content;
    
  }
