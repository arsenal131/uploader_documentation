async workStatus()

Background

Work status is important in work flow to determine if previous work flow completed or did not. It dictates new work flow or continuation of previous work flow.

Objectives

To read workStatus file.

Flowchart

Code

  async workStatus(){

    /*************************+**/
    /* To read workStatus file */
    /**************************+*/
  
    const ws = _FS.promises.readFile( _WS_PATH, "utf-8" );
    
    this.res = await ws
      .then( 
        async ( resolved )=>{ return resolved }, 
        async ( rejected )=>{
          echo( "\n# workStatus data: not set" );
          echo( "#rejected: ", rejected );
          return rejected;
        }
      )
      .catch( async ( error )=> {
        if( error ){
          echo( `\n# workStatus error: ${error}` );
          return false;
        }
      } )
      .finally( 
        async function(){
          echo( "# workStatus delivered" );
        }.bind(this) 
      )
    
    return this;
    
  }
