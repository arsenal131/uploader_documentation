  async set_author( page, author ){
    
    echo( "running set_author method" );
    
    echo( "focus on element #author" );
    
    await page.focus( "#author" );
    
    echo( "typing author name" );
    
    await page.keyboard.type( author );
    
    echo( "press tab" );
    
    await page.keyboard.press( "Tab" );
    
    return page;
    
  }
