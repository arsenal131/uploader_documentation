  async set_category ( page, category_id ){
  
    echo( "change post category to journal" );
    
    var scp = await page.evaluate( ( category_id ) => { 
    
      let event = new Event( 'change', { bubbles: true } );
      
      let element = document.querySelector( '#post_category' );
      
      event.simulated=true;
      
      document.querySelector( `#post_category > option:nth-child( ${category_id} )` ).selected = true; // 1 -> journal
      
      let de = element.dispatchEvent( event );
      
      return de;
      
    }, category_id );
    
    return scp;
    
  }
