  async set_first_page_id ( page, first_page_id ){
    
    echo("running set_first_page_id method");
    
    await page.evaluate( ( first_page_id ) => { 
    
      var is_serial_post = document.getElementById( 'is_serial_post' );
      
      if( !!is_serial_post ){
      
        if( !is_serial_post.checked ){
        
          is_serial_post.click( );
          
        }
        
      }
      
    }, first_page_id );
    
    echo( `first_page_id: ${!!first_page_id}`,  );
    
    if( !!first_page_id ){
    
      echo( `first_page_id: ${first_page_id}` );
      
      echo( "focus on #serial_tag" );  
    
      await page.focus( '#serial_tag' );
    
      echo( `pass first_page_id ${first_page_id} to #serial_tag` );
    
      await page.keyboard.type( first_page_id );
    
      return true;
    
    } else if( !first_page_id ){
    
      echo( "get first_page_id from #id_post" );
    
      first_page_id = await page.$eval( "#id_post", el => el.value );
    
      do{
      
        first_page_id = await page.$eval( "#id_post", el => el.value );
        
        echo( 'first_page_id: ', first_page_id );
        
        await page.waitForTimeout( 1000 );
        
      }while( !first_page_id );
    
      echo( `first_page_id: ${first_page_id}` );
      
      echo( "focus on #serial_tag" );
      
      await page.focus( '#serial_tag' );
      
      echo( `pass first_page_id ${first_page_id} to #serial_tag` );
      
      await page.keyboard.type( first_page_id );
      
      return true;
      
    }
    
    return false;
    
  }
