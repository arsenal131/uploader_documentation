  async editorTextArea ( page, content ){
    
    echo( "editorTextArea method starts" );
    
    const cnt = await page.evaluate( 
      async ( content ) => { 
      
        let edtr = document.getElementById( "editor" );
        edtr.value = content;
        
        return edtr.value;
        
      }, 
      content
    );    
    
    return cnt;
    
  }
