Array.prototype.loop = async function( i = 0, callback ){
  
  var i_length = this.length;
  var arrs = this;
  
  var arr = [];
  
  async function loop_run( i, i_length, arrs ){
    
    let cb = await callback( i, arrs[i] );
    
    if( cb ){
      arr.push( [cb, parseInt(cb.split("_").reverse()[0]) ] );
    }
    
    i++;
    if( i < i_length ){
      return loop_run( i, i_length, arrs );
    }
    
    return i;
    
  }
  
  let lr = await loop_run(0, i_length, arrs);
  
  return arr.sort( ( a, b ) => a[1] - b[1] ).map( L => L[0]  );
  
}
