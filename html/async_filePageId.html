  async filePageId( page, path, firstPageId ){ //

    echo( "filePageId starts" );
      
    var currentPageId, retWriteFileFirst, retWriteFileCurrent = null; //

    // get first page id
    path = _WORKING_DIRECTORY + path; //
    
    if( !firstPageId ){ //
      
      echo( "get #id_post value of first page" ); //
      
      do{
        
        echo( "trying to obtain firstPageId" ); //
        
        await page.waitForSelector( "#id_post", { timeout:10000 } );  //
        
        firstPageId = await page.$eval( "#id_post", async el => el.value ); //
        
        if( firstPageId > 0 ){  //
        
          echo( "firstPageId: " , firstPageId ); //
          
          echo( "save firstPageId to file" ); //
          
          if( !_FS.existsSync( path ) ){  //
          
            _FS.mkdirSync( path );  //
            
          }
          
          var retWriteFileFirst = await _FS.writeFile( 
            path + "/firstPageId", 
            firstPageId, 
            async ( err )=>{
            
              if ( err ){
              
                throw err;
                
                return false;
                
              } else {
              
                return true;
                
              }
              
            } 
          );
          
          echo( "retWriteFileFirst: ", retWriteFileFirst );
          
          echo( "save currentPageId to file" );
          
          currentPageId = firstPageId;
          
          retWriteFileCurrent = await _FS.writeFile( 
            path + "/currentPageId", 
            currentPageId, 
            async ( err )=>{
            
              if ( err ){
              
                throw err;
                
                return false;
              
              } else {
                
                return true;
              
              }
            
            } 
          ); 
          
          echo( "retWriteFileCurrent: ", retWriteFileCurrent );
          
          return currentPageId;
            
        }
        
        await page.waitForTimeout( 1000 );
        
      } while( !firstPageId );

      return firstPageId;      
      
    } else {
      
      echo( "resave firstPageId" );
      
      retWriteFileFirst = await _FS.writeFile( 
        path + "/firstPageId", 
        firstPageId, 
        async ( err )=>{
          
          if ( err ){
            
            echo( err );
            
            return false;
          
          } else {
            
            return true;
          
          }
          
        }
        
      );
      
      echo( "retWriteFileFirst: ", retWriteFileFirst );
      
      echo( "get #id_post value of current page" );
      
      await page.waitForSelector( "#id_post", { timeout:10000 } );
      
      currentPageId = await page.$eval( "#id_post", async el => el.value );
      
      echo( `currentPageId: ${currentPageId}` );
      
      echo( "save currentPageId to file" );
      
      if( currentPageId > 0 ){     //
      
        retWriteFileCurrent = await _FS.writeFile( 
          path + "/currentPageId", 
          currentPageId, 
          async ( err )=>{
          
            if ( !!err ){
              
              throw err;
              
              return false;
              
            } else {
              
              return true;
              
            }
            
          } 
        
        );
        
      }
      
      do{
        
        echo( "retrying to obtain currentPageId" );
        
        await page.waitForSelector( "#id_post", { timeout:10000 } );
        
        currentPageId = await page.$eval( "#id_post", async el => el.value );
        
        if( currentPageId > 0 ){
        
          retWriteFileCurrent = await _FS.writeFile( 
            path + "/currentPageId", 
            currentPageId, 
            async ( err )=>{
              
              if ( err ){
              
                echo( err );
              
                return false;
              
              } else {
              
                return true;
              
              }
              
            }
            
          );
          
          echo( retWriteFileCurrent );
          
        }
        
        await page.waitForTimeout( 1000 );          
      
      } while( !currentPageId );
      
      return currentPageId;
      
    }
    
  }
