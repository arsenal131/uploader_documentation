  async setPublishStatus ( page, statusID ){
    
    echo( "setPublishStatus method starts" );
    
    echo( `set publish status to ${status_ID}` );
    
    await page.evaluate( async ( status_ID )=>{ 
    
      let event = new Event( 'change', { bubbles: true } );
      let element = document.querySelector( '#status' );
      let dqs = document.querySelector( `#status > option:nth-child( ${status_ID} )` );

      event.simulated=true;

      dqs.selected = true;
      
      element.dispatchEvent( event );
      
      return true;
      
    }, status_ID );
    
  }
