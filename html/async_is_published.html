  async is_published( page, current_page_id, pretty_link ){

    // method 1: go to publish box
    echo( "method 1: go to publish box" );
    echo( `check #${current_page_id} in publish_box` );
        
    let pub_box = "https://palmerah.implantgigi.com/insight/edit/list/2";
    echo( `go to ${pub_box}` );
    await page.goto( pub_box, {waitUntil:'domcontentloaded'} );
  
    let elm = `tr#record_${ current_page_id }`;
      
    echo( `searching for element ${elm}` );
    let is_elm = await page.$$( elm );
    
    // method 2: go to web page
    
    echo( "method 2: go to web page" );
    
    echo( `check if post #${current_page_id} published in live web page` );
    
    echo( `go to https://palmerah.implantgigi.com/insight/${pretty_link}` );

    let live_uri = "https://palmerah.implantgigi.com/insight/" + pretty_link;

    await this.goto_uri( page, live_uri );

    let elm_post_body = "div.post-body.mt-5";
      
    echo( `searching for element ${elm_post_body}` );

    let ret_elm_post_body = await page.evaluate( 
      async function( elm_post_body ){
        let htmle = document.querySelector(  `${elm_post_body}` ).innerHTML;
        return htmle;
      },
      elm_post_body 
    );
    
    echo( "variable ret_elm_post_body: ", !!ret_elm_post_body );
    
    if( !!is_elm && ret_elm_post_body.length > 10 ){
      return elm;
    } else if( !is_elm || !ret_elm_post_body ){
      return null;
    }

  }
