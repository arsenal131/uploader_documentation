  async savePrettyLink( page, prettyLink ){
  
      echo( "savePrettyLink method starts" );
      
      let savePrettyLinkToFile = await _FS.promises.appendFile( PATH_PRETTY_LINKS_FILE, prettyLink + "\n")
        .then( 
          async err => !err
        );

      echo( `link saved as ${PATH_PRETTY_LINKS_FILE}` );

      return savePrettyLinkToFile;
  }
