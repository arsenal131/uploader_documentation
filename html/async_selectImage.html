  async selectImage( page, idImage ){
  
    echo( "selectImage method starts" );

    var wSelector = false;
  
    var frame, el, selektor = null;
    
    var frmSrc = `iframe[src="${_IMG_IDX}"]`;

    do{

      await page.focus( '#mceu_15-button' );
      await page.click( '#mceu_15-button' );
    
      el = await page.$( frmSrc );
  
      frame = await el.contentFrame();
  
      selektor = "input#rb_" + idImage;
  
      echo( "selektor: " , selektor );
  
      wSelector = await frame.waitForSelector( selektor )
        .then( 
          async ()=>{
            
            echo( `${selektor} found` );
            
            wSelector = true;
            
            return true;
            
          } 
        )
        .catch( 
          async ( err )=>{
            
            echo( `${selektor} not found` );
            
            wSelector = false;
            
            return false;
            
          }
        );
      
      await frame.waitForTimeout( 2000 );
      
    } while( !wSelector );

    await page.waitForTimeout( 5000 );
    
    await frame.evaluate( 
    
      async ( selektor ) => {
    
        let t = document.querySelector( selektor );
        
        t.classList.remove( "d-none" );
        
        t.click();
        
        t.parentNode;
        
        t.parentNode.click();
        
      }, 
      selektor 
    );
  
    await page.waitForTimeout( 10000 );
    
    page.on( "response", async ( data )=>{
      
      echo( "response: ", data );
      
      return data;
      
    } );

    echo( "click insert image button" );
    
    var res = await page.evaluate( async () => {
    
      let btns = document.querySelectorAll( "button" );
      
      if( !btns ){
      
        throw "variable btns is empty";
        
        return;
      
      }
      
      const cbh = [].map.call( btns, async ( btn )=>{
      
        let bitt = btn.innerText.trim();
      
        if( bitt == "Ok" ){
        
          btn.click();
          
          return true;
          
        } else {
        
          return false
          
        }
      
      } );
      
      return cbh;
    
    } );
    
    return res;
  
  }
