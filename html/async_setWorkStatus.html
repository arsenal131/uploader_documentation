    echo( `# setting work_status to ${status} ... ` );

      .then( ( err )=>{

        if( err ) {
          echo( err );
          return err;
        }
        
        let msg = `OK, work_status changed to ${status}`;
        
        return msg;
        
      } );
      
    return fsp;
    
  }
