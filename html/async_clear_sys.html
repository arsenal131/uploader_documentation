  async clear_sys( path ){
    
    /***************************/
    /* To delete working files */
    /***************************/
    
    echo( "# clearing working files start" );
    
    var files_to_delete = [
      "/home/bisnis/node/id_image", 
      `/home/bisnis/node/${path}/first_page_id`, 
      `/home/bisnis/node/${path}/current_page_id`
    ];
    
    let iterations = files_to_delete.length;
    
    await files_to_delete.loop(0, async function ( i, file ){ 
      
      await _FS.promises.unlink( file )
        .then(
          async ( resolved )=>{ return resolved }, 
          async ( err )=>{  
          
            if( err && err.code == 'ENOENT' ){
              echo( `# ${wfn}. ${file} does not exist` );
              wfn++;
              this.return_message = `# ${wfn}. ${file} does not exist`;
              return false;
            } else if( err ){
              //console.log( "# error occured while trying remove file" );
              this.return_message = "# error occured while trying remove file";
              return false;
            } else if( !err ){
              //console.log( `# ${wfn}. ${file} removed` )
              wfn++;
              this.return_message = `# ${wfn}. ${file} removed`;
              return true;
            }
          
          } 
        )
        .catch( 
          async ( err )=>{ 
            
            echo( err ); 
            
            return err;
            
          } 
        )
        .finally( async ()=>{
          this.clear_sys_end = true;
        } )
      
      
    }.bind( this ) );
    
    return this;

  }
