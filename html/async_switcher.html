  async switcher( data ){
  
    var ret = false;
  
    switch( data ){
      case "on-going":
        echo( "# resume previous process" );
        ret = await this.resume_previous_process( path );
        return ret;
        break;
      case "finish":
        echo( "# previous process finished, start new process" );
        ret = await this.promise_all_files( path );
        return ret;
        break;
      default:
        echo( "# unknown work status" );
        return ret;
        break;
    }

  }
