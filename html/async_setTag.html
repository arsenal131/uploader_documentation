  async setTag( page, tags ){

    echo( "setTag method starts" );
    await page.focus( '[name="tags"]' );

    echo( `typing tags ${tags}` );
    await page.keyboard.type( tags );

    echo( "press tab key" );
    await page.keyboard.press( "Tab" );

    return page;
    
  }
