  async file_page_id( page, path, first_page_id ){
    echo( "put first page id into file" );
      
    var current_page_id = "";
    var ret_write_file_first = null;
    var ret_write_file_current = null;
      
    // get first page id
    path = "/home/bisnis/node/" + path;
    if( !first_page_id ){
      echo( "get #id_post value of first page" );
      do{
        echo( "trying to obtain first_page_id" );
        await page.waitForSelector( "#id_post", { timeout:10000 } );
        first_page_id = await page.$eval( "#id_post", el => el.value );
        if( first_page_id > 0 ){
          echo( "first page id: " , first_page_id );
          echo( "save first page id to file first_page_id" );
          if( !_FS.existsSync( path ) ){
            _FS.mkdirSync( path );
          }
          var ret_write_file_first = await _FS.writeFile( path + "/first_page_id", first_page_id, function( err ){
            if ( err ){
              echo( err );
              return false;
            } else {
              return true;
            }
          } );
          echo( "file writing status of first: ," ret_write_file_first );
          echo( "save current page id to file current_page_id" );
          current_page_id = first_page_id;
          ret_write_file_current = await _FS.writeFile( path + "/current_page_id", current_page_id, async function( err ){
            if ( err ){
              echo( err );
              return false;
            } else {
              return true;
            }
          } ); 
          echo( "file writing status of current: ", ret_write_file_current )
          return current_page_id;
            
        }
        await page.waitForTimeout( 1000 );
      }
      while( !first_page_id );
      return first_page_id;
    } else {
      echo( "resave first_page_id into file to ensure" );
      ret_write_file_first = await _FS.writeFile( path + "/first_page_id", first_page_id, async function( err ){
        if ( err ){
          echo( err );
          return false;
        } else {
          return true;
        }
      });
      echo( "file writing status of first", ret_write_file_first );
      echo( "get #id_post value of current page" );
      await page.waitForSelector( "#id_post", { timeout:10000 } );
      current_page_id = await page.$eval( "#id_post", el => el.value );
      echo( `current page id: ${current_page_id}` );
      echo( "save current page id to file current_page_id" );
      if( current_page_id > 0 ){
        ret_write_file_current = await _FS.writeFile( path + "/current_page_id", current_page_id, async function( err ){
          if ( err ){
            echo( err );
            return false;
          } else {
            return true;
          }
        } );
      }
      do{
        echo( "retrying to obtain current_page_id" );
        await page.waitForSelector( "#id_post", { timeout:10000 } );
        current_page_id = await page.$eval( "#id_post", el => el.value );
        if( current_page_id > 0 ){
          ret_write_file_current = await _FS.writeFile( path + "/current_page_id", current_page_id, async function( err ){
            if ( err ){
              echo( err );
              return false;
            } else {
              return true;
            }
          } );
          echo( ret_write_file_current );
        }
        await page.waitForTimeout( 1000 );          
      }while( !current_page_id )
      return current_page_id;
    }
  }
