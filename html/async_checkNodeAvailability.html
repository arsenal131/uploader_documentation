  async checkNodeAvailability ( page, nodeIdentity ){
  
    //to ensure availability nodes before processing  
    echo( `check if node ${nodeIdentity} are available` );    

    var elImgTitle = false;
    var nEit = 1;
                    
    do{
      elImgTitle = await page.waitForSelector( node_identity, { timeout:20000 } )
        .then( 
          async ( resolved )=>{
            echo( `element #img_title found in ${nEit} attempts` );
            return resolved;
          }
        )
        .catch( 
          async ( err )=>{
            if( err ){
              return false;
            }
          } 
        );
      await page.waitForTimeout( 1000 );
      nEit++;
    }while( !elImgTitle );
      
    return elImgTitle;
      
  }
