  async title( page, title ){
      
      if( title.length > 200 ){
      
        title = title.substr( 0,100 );
        
        title = title.substr( 0, title.lastIndexOf( " " ) );
        
      }
      
      // set post title
      
      echo( "set post title: ", title );
      
      //await page.evaluate( ( title ) => {document.getElementById( "input_title" ).value = title }, title );
      
      await page.focus( "#input_title" );
      
      echo( "typing title" );
      
      await page.keyboard.type( title );
      
      echo( "enter tab button" );
      
      await page.keyboard.press( "Tab" );
      
      return page;
      
  }
