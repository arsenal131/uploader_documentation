Flowchart

Purpose

To delete working files.

Code

  async clearSys( path ){
    
    /***************************/
    /* To delete working files */
    /***************************/
    
    echo( "# clearing working files start" );
    
    const filesToDelete = [
      "/home/bisnis/node/id_image", 
      `/home/bisnis/node/${path}/first_page_id`, 
      `/home/bisnis/node/${path}/current_page_id`
    ];
    
    let iterations = filesToDelete.length;
    
    await filesToDelete.loop(0, async function (i, file){ 
      
      await _FS.promises.unlink( file )
        .then(
          async ( resolved )=>{ return resolved }, 
          async ( err )=>{  
          
            if( err && err.code == 'ENOENT' ){
              echo( `# ${wfn}. ${file} does not exist` );
              wfn++;
              this.returnMessage = `# ${wfn}. ${file} does not exist`;
              return false;
            } else if( err ){
              this.returnMessage = "# error occured while trying remove file";
              return false;
            } else if( !err ){
              wfn++;
              this.returnMessage = `# ${wfn}. ${file} removed`;
              return true;
            }
          
          } 
        )
        .catch( async ( err )=>{ echo( err ); } )
        .finally( async ()=>{
          this.clearSysEnd = true;
        } )
      
      
    }.bind( this ) );
    
    return this;

  }
